import { useEffect, useState } from "react";
import { motion } from "framer-motion";
import {
  BrainCircuit,
  CheckCircle2,
  ChevronDown,
  ChevronUp,
  FolderOpen,
  Gauge,
  GitBranch,
  Package,
  RefreshCw,
  Sparkles,
  XCircle
} from "lucide-react";

interface Project {
  id: number;
  name: string;
  localPath: string;
  packageManager: string | null;
  detectedStack: string[];
  scripts: Record<string, string>;
  readinessScore: number;
  readinessReport: {
    score: number;
    checks: {
      key: string;
      label: string;
      passed: boolean;
      points: number;
      message: string;
    }[];
    issues: string[];
  };
  createdAt: string;
  updatedAt: string;
  lastScanAt: string | null;
}

interface TaskPack {
  id: number;
  projectId: number;
  projectName?: string;
  title: string;
  rawTask: string;
  taskType: string;
  targetTool: string;
  generatedPrompt: string;
  createdAt: string;
  updatedAt: string;
}

const API_URL = "http://localhost:4000/api";

function getScoreLabel(score: number) {
  if (score >= 80) return "Good";
  if (score >= 60) return "Acceptable";
  if (score >= 40) return "Risky";
  return "Poor";
}

function App() {
  const [projects, setProjects] = useState<Project[]>([]);
  const [expandedProjectId, setExpandedProjectId] = useState<number | null>(null);
  const [isLoading, setIsLoading] = useState(false);
  const [statusMessage, setStatusMessage] = useState("Ready to scan your first project.");
  const [agentsPreview, setAgentsPreview] = useState<{
    projectId: number;
    projectName: string;
    markdown: string;
  } | null>(null);
  const [taskPacks, setTaskPacks] = useState<TaskPack[]>([]);
  const [taskPackDraft, setTaskPackDraft] = useState<{
    projectId: number;
    projectName: string;
    rawTask: string;
    taskType: string;
    targetTool: string;
  } | null>(null);
  const [generatedTaskPack, setGeneratedTaskPack] = useState<TaskPack | null>(null);

  async function loadProjects() {
    const response = await fetch(`${API_URL}/projects`);
    const data = await response.json();

    if (data.ok) {
      setProjects(data.projects);
    }
  }

  async function loadTaskPacks() {
    const response = await fetch(`${API_URL}/task-packs`);
    const data = await response.json();

    if (data.ok) {
      setTaskPacks(data.taskPacks);
    }
  }

  async function handleSelectProject() {
    const selectedPath = await window.contextforge?.selectProjectFolder?.();

    if (!selectedPath) {
      return;
    }

    try {
      setIsLoading(true);
      setStatusMessage("Scanning project...");

      const response = await fetch(`${API_URL}/projects`, {
        method: "POST",
        headers: {
          "Content-Type": "application/json"
        },
        body: JSON.stringify({
          localPath: selectedPath
        })
      });

      const data = await response.json();

      if (!data.ok) {
        throw new Error(data.message ?? "Failed to add project");
      }

      await loadProjects();
      setExpandedProjectId(data.project.id);
      setStatusMessage(`Project "${data.project.name}" added successfully.`);
    } catch (error) {
      setStatusMessage(error instanceof Error ? error.message : "Unknown error");
    } finally {
      setIsLoading(false);
    }
  }

  async function handleRescanProject(project: Project) {
    try {
      setIsLoading(true);
      setStatusMessage(`Rescanning "${project.name}"...`);

      const response = await fetch(`${API_URL}/projects/${project.id}/rescan`, {
        method: "POST"
      });

      const data = await response.json();

      if (!data.ok) {
        throw new Error(data.message ?? "Failed to rescan project");
      }

      await loadProjects();
      setExpandedProjectId(project.id);
      setStatusMessage(`Project "${project.name}" rescanned successfully.`);
    } catch (error) {
      setStatusMessage(error instanceof Error ? error.message : "Unknown error");
    } finally {
      setIsLoading(false);
    }
  }

  async function handleGenerateAgentsPreview(project: Project) {
    try {
      setIsLoading(true);
      setStatusMessage(`Generating AGENTS.md for "${project.name}"...`);

      const response = await fetch(`${API_URL}/projects/${project.id}/agents-preview`);
      const data = await response.json();

      if (!data.ok) {
        throw new Error(data.message ?? "Failed to generate AGENTS.md preview");
      }

      setAgentsPreview({
        projectId: project.id,
        projectName: project.name,
        markdown: data.markdown
      });

      setStatusMessage(`AGENTS.md preview generated for "${project.name}".`);
    } catch (error) {
      setStatusMessage(error instanceof Error ? error.message : "Unknown error");
    } finally {
      setIsLoading(false);
    }
  }

  async function handleSaveAgentsFile() {
    if (!agentsPreview) {
      return;
    }

    try {
      setIsLoading(true);
      setStatusMessage(`Saving AGENTS.md for "${agentsPreview.projectName}"...`);

      const response = await fetch(
        `${API_URL}/projects/${agentsPreview.projectId}/agents-save`,
        {
          method: "POST"
        }
      );

      const data = await response.json();

      if (!data.ok) {
        throw new Error(data.message ?? "Failed to save AGENTS.md");
      }

      await loadProjects();
      setStatusMessage(`AGENTS.md saved to project "${agentsPreview.projectName}".`);
      setAgentsPreview(null);
    } catch (error) {
      setStatusMessage(error instanceof Error ? error.message : "Unknown error");
    } finally {
      setIsLoading(false);
    }
  }

  async function handleCreateTaskPack() {
    if (!taskPackDraft) {
      return;
    }

    try {
      setIsLoading(true);
      setStatusMessage(`Generating task pack for "${taskPackDraft.projectName}"...`);

      const response = await fetch(`${API_URL}/task-packs`, {
        method: "POST",
        headers: {
          "Content-Type": "application/json"
        },
        body: JSON.stringify({
          projectId: taskPackDraft.projectId,
          rawTask: taskPackDraft.rawTask,
          taskType: taskPackDraft.taskType,
          targetTool: taskPackDraft.targetTool
        })
      });

      const data = await response.json();

      if (!data.ok) {
        throw new Error(data.message ?? "Failed to generate task pack");
      }

      await loadTaskPacks();
      setGeneratedTaskPack(data.taskPack);
      setTaskPackDraft(null);
      setStatusMessage("Task pack generated successfully.");
    } catch (error) {
      setStatusMessage(error instanceof Error ? error.message : "Unknown error");
    } finally {
      setIsLoading(false);
    }
  }

  useEffect(() => {
    Promise.all([loadProjects(), loadTaskPacks()]).catch(() => {
      setStatusMessage("Failed to load initial data.");
    });
  }, []);

  const readinessScore =
    projects.length > 0
      ? Math.round(
        projects.reduce((total, project) => total + project.readinessScore, 0) /
        projects.length
      )
      : null;

  return (
    <main className="flex h-screen w-screen overflow-hidden bg-black text-neutral-100">
      <aside className="flex w-72 flex-col border-r border-neutral-900 bg-black p-5">
        <div className="mb-10 flex items-center gap-3">
          <div className="flex size-10 items-center justify-center rounded-xl border border-neutral-800 bg-neutral-950 text-white">
            <BrainCircuit size={21} />
          </div>

          <div>
            <h1 className="text-[15px] font-semibold tracking-tight text-white">
              ContextForge
            </h1>
            <p className="text-xs text-neutral-500">AI project control center</p>
          </div>
        </div>

        <nav className="space-y-1 text-sm">
          {["Dashboard", "Projects", "Context Builder", "Task Packs", "Settings"].map(
            (item, index) => (
              <button
                key={item}
                className={[
                  "w-full rounded-lg px-3 py-2.5 text-left transition",
                  index === 0
                    ? "bg-neutral-900 text-white"
                    : "text-neutral-400 hover:bg-neutral-950 hover:text-white"
                ].join(" ")}
              >
                {item}
              </button>
            )
          )}
        </nav>

        <div className="mt-auto rounded-2xl border border-neutral-900 bg-neutral-950/60 p-4">
          <p className="mb-2 text-[11px] uppercase tracking-[0.22em] text-neutral-600">
            MVP Status
          </p>
          <p className="text-sm text-neutral-300">Phase 0.1 — Project Scanner</p>
        </div>
      </aside>

      <section className="flex flex-1 flex-col bg-black">
        <header className="app-drag flex h-16 items-center justify-between border-b border-neutral-900 px-8">
          <div>
            <p className="text-xs text-neutral-500">Welcome back</p>
            <h2 className="text-lg font-semibold tracking-tight text-white">Dashboard</h2>
          </div>

          <button
            onClick={handleSelectProject}
            disabled={isLoading}
            className="app-no-drag cf-button-primary"
          >
            <FolderOpen size={16} />
            {isLoading ? "Scanning..." : "Add project"}
          </button>
        </header>

        <div className="flex-1 overflow-auto p-8">
          <motion.div
            initial={{ opacity: 0, y: 14 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ duration: 0.35 }}
            className="mb-8"
          >
            <p className="cf-badge mb-4">
              <Sparkles size={13} />
              Local-first AI workflow manager
            </p>

            <h3 className="max-w-4xl text-[44px] font-semibold leading-[1.03] tracking-[-0.045em] text-white">
              Prepare your projects for Codex, Cursor, Claude Code and other AI agents.
            </h3>

            <p className="mt-5 max-w-2xl text-[15px] leading-6 text-neutral-400">
              ContextForge scans your project, builds AI-ready context, generates task packs,
              and helps prevent AI agents from breaking your architecture.
            </p>
          </motion.div>

          <div className="mb-8 grid grid-cols-3 gap-4">
            <div className="cf-card p-6">
              <div className="mb-5 flex size-10 items-center justify-center rounded-xl border border-neutral-800 bg-neutral-950 text-neutral-200">
                <Gauge size={20} />
              </div>
              <p className="text-sm text-neutral-500">AI Readiness</p>
              <p className="mt-2 text-3xl font-semibold tracking-tight text-white">
                {readinessScore ?? "—"}
              </p>
              {readinessScore !== null && (
                <p className="mt-2 text-sm text-neutral-500">
                  {getScoreLabel(readinessScore)}
                </p>
              )}
            </div>

            <div className="cf-card p-6">
              <div className="mb-5 flex size-10 items-center justify-center rounded-xl border border-neutral-800 bg-neutral-950 text-neutral-200">
                <Package size={20} />
              </div>
              <p className="text-sm text-neutral-500">Projects</p>
              <p className="mt-2 text-3xl font-semibold tracking-tight text-white">
                {projects.length}
              </p>
            </div>

            <div className="cf-card p-6">
              <div className="mb-5 flex size-10 items-center justify-center rounded-xl border border-neutral-800 bg-neutral-950 text-neutral-200">
                <GitBranch size={20} />
              </div>
              <p className="text-sm text-neutral-500">Task Packs</p>
              <p className="mt-2 text-3xl font-semibold tracking-tight text-white">
                {taskPacks.length}
              </p>
            </div>
          </div>

          <div className="mb-6 rounded-xl border border-neutral-900 bg-neutral-950/50 px-4 py-3 text-sm text-neutral-400">
            {statusMessage}
          </div>

          <section>
            <div className="mb-4 flex items-center justify-between">
              <div>
                <h4 className="text-base font-medium text-white">Projects</h4>
                <p className="mt-1 text-sm text-neutral-500">
                  Scanned local repositories and detected AI-ready context.
                </p>
              </div>
            </div>

            {projects.length === 0 ? (
              <div className="cf-card flex min-h-52 flex-col items-center justify-center p-8 text-center">
                <div className="mb-5 flex size-12 items-center justify-center rounded-2xl border border-neutral-800 bg-neutral-950 text-neutral-300">
                  <FolderOpen size={22} />
                </div>

                <h5 className="text-lg font-medium text-white">No projects yet</h5>
                <p className="mt-2 max-w-md text-sm leading-6 text-neutral-500">
                  Add your first local project to scan its structure, detect stack,
                  and prepare it for AI agents.
                </p>

                <button
                  onClick={handleSelectProject}
                  disabled={isLoading}
                  className="cf-button-secondary mt-6"
                >
                  <FolderOpen size={16} />
                  Select folder
                </button>
              </div>
            ) : (
              <div className="grid gap-4">
                {projects.map((project) => {
                  const isExpanded = expandedProjectId === project.id;

                  return (
                    <article key={project.id} className="cf-card p-5">
                      <div className="flex items-start justify-between gap-6">
                        <div className="min-w-0">
                          <div className="mb-2 flex items-center gap-2">
                            <CheckCircle2 size={16} className="text-neutral-300" />
                            <h5 className="truncate text-base font-medium text-white">
                              {project.name}
                            </h5>
                          </div>

                          <p className="truncate text-sm text-neutral-500">
                            {project.localPath}
                          </p>

                          <div className="mt-4 flex flex-wrap gap-2">
                            {project.detectedStack.map((item) => (
                              <span key={item} className="cf-badge">
                                {item}
                              </span>
                            ))}
                          </div>
                        </div>

                        <div className="flex shrink-0 items-start gap-8">
                          <div className="text-right">
                            <p className="text-xs uppercase tracking-[0.18em] text-neutral-600">
                              AI Ready
                            </p>
                            <p className="mt-2 text-lg font-semibold text-white">
                              {project.readinessScore}/100
                            </p>

                            <p className="mt-4 text-xs uppercase tracking-[0.18em] text-neutral-600">
                              Package manager
                            </p>
                            <p className="mt-2 text-sm text-neutral-300">
                              {project.packageManager ?? "Unknown"}
                            </p>
                          </div>

                          <div className="flex flex-col gap-2">
                            <button
                              onClick={() => handleGenerateAgentsPreview(project)}
                              disabled={isLoading}
                              className="rounded-lg border border-neutral-800 bg-neutral-950 px-3 py-2 text-xs text-neutral-300 transition hover:border-neutral-700 hover:text-white"
                              title="Generate AGENTS.md"
                            >
                              AGENTS
                            </button>

                            <button
                              onClick={() =>
                                setTaskPackDraft({
                                  projectId: project.id,
                                  projectName: project.name,
                                  rawTask: "",
                                  taskType: "general",
                                  targetTool: "codex"
                                })
                              }
                              disabled={isLoading}
                              className="rounded-lg border border-neutral-800 bg-neutral-950 px-3 py-2 text-xs text-neutral-300 transition hover:border-neutral-700 hover:text-white"
                              title="Create task pack"
                            >
                              TASK
                            </button>


                            <button
                              onClick={() => handleRescanProject(project)}
                              disabled={isLoading}
                              className="rounded-lg border border-neutral-800 bg-neutral-950 p-2 text-neutral-400 transition hover:border-neutral-700 hover:text-white"
                              title="Rescan project"
                            >
                              <RefreshCw size={15} />
                            </button>

                            <button
                              onClick={() =>
                                setExpandedProjectId(isExpanded ? null : project.id)
                              }
                              className="rounded-lg border border-neutral-800 bg-neutral-950 p-2 text-neutral-400 transition hover:border-neutral-700 hover:text-white"
                              title="Toggle readiness report"
                            >
                              {isExpanded ? <ChevronUp size={15} /> : <ChevronDown size={15} />}
                            </button>
                          </div>
                        </div>
                      </div>

                      {isExpanded && (
                        <motion.div
                          initial={{ opacity: 0, height: 0 }}
                          animate={{ opacity: 1, height: "auto" }}
                          transition={{ duration: 0.22 }}
                          className="mt-5 overflow-hidden border-t border-neutral-900 pt-5"
                        >
                          <div className="grid gap-3 md:grid-cols-2">
                            {project.readinessReport.checks.map((check) => (
                              <div
                                key={check.key}
                                className="rounded-xl border border-neutral-900 bg-black/40 p-4"
                              >
                                <div className="flex items-start gap-3">
                                  {check.passed ? (
                                    <CheckCircle2
                                      size={17}
                                      className="mt-0.5 text-emerald-400"
                                    />
                                  ) : (
                                    <XCircle
                                      size={17}
                                      className="mt-0.5 text-neutral-600"
                                    />
                                  )}

                                  <div>
                                    <div className="flex items-center gap-2">
                                      <p className="text-sm font-medium text-white">
                                        {check.label}
                                      </p>
                                      <span className="text-xs text-neutral-600">
                                        {check.passed ? `+${check.points}` : `0/${check.points}`}
                                      </span>
                                    </div>

                                    <p className="mt-1 text-sm leading-5 text-neutral-500">
                                      {check.passed
                                        ? check.message
                                        : "Missing or not detected."}
                                    </p>
                                  </div>
                                </div>
                              </div>
                            ))}
                          </div>

                          {project.readinessReport.issues.length > 0 && (
                            <div className="mt-4 rounded-xl border border-neutral-900 bg-black/40 p-4">
                              <p className="mb-3 text-sm font-medium text-white">
                                Recommended improvements
                              </p>

                              <ul className="space-y-2">
                                {project.readinessReport.issues.map((issue) => (
                                  <li
                                    key={issue}
                                    className="text-sm leading-5 text-neutral-500"
                                  >
                                    • {issue}
                                  </li>
                                ))}
                              </ul>
                            </div>
                          )}
                        </motion.div>
                      )}
                    </article>
                  );
                })}
              </div>
            )}
          </section>
        </div>
      </section>
      {agentsPreview && (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/70 p-8 backdrop-blur-sm">
          <div className="flex max-h-[82vh] w-full max-w-5xl flex-col overflow-hidden rounded-2xl border border-neutral-800 bg-black shadow-2xl">
            <div className="flex items-center justify-between border-b border-neutral-900 px-5 py-4">
              <div>
                <p className="text-xs uppercase tracking-[0.18em] text-neutral-600">
                  Context Builder
                </p>
                <h3 className="mt-1 text-base font-medium text-white">
                  AGENTS.md preview — {agentsPreview.projectName}
                </h3>
              </div>

              <button
                onClick={() => setAgentsPreview(null)}
                className="rounded-lg border border-neutral-800 bg-neutral-950 px-3 py-2 text-sm text-neutral-400 transition hover:text-white"
              >
                Close
              </button>
            </div>

            <pre className="flex-1 overflow-auto p-5 text-sm leading-6 text-neutral-300">
              {agentsPreview.markdown}
            </pre>

            <div className="flex items-center justify-end gap-3 border-t border-neutral-900 px-5 py-4">
              <button
                onClick={() => navigator.clipboard.writeText(agentsPreview.markdown)}
                className="cf-button-secondary"
              >
                Copy
              </button>

              <button
                onClick={handleSaveAgentsFile}
                disabled={isLoading}
                className="cf-button-primary"
              >
                Save to project
              </button>
            </div>
          </div>
        </div>
      )}

      {taskPackDraft && (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/70 p-8 backdrop-blur-sm">
          <div className="flex w-full max-w-3xl flex-col overflow-hidden rounded-2xl border border-neutral-800 bg-black shadow-2xl">
            <div className="flex items-center justify-between border-b border-neutral-900 px-5 py-4">
              <div>
                <p className="text-xs uppercase tracking-[0.18em] text-neutral-600">
                  Task Pack Builder
                </p>
                <h3 className="mt-1 text-base font-medium text-white">
                  New task for {taskPackDraft.projectName}
                </h3>
              </div>

              <button
                onClick={() => setTaskPackDraft(null)}
                className="rounded-lg border border-neutral-800 bg-neutral-950 px-3 py-2 text-sm text-neutral-400 transition hover:text-white"
              >
                Close
              </button>
            </div>

            <div className="space-y-4 p-5">
              <div>
                <label className="mb-2 block text-sm text-neutral-400">
                  Task description
                </label>
                <textarea
                  value={taskPackDraft.rawTask}
                  onChange={(event) =>
                    setTaskPackDraft({
                      ...taskPackDraft,
                      rawTask: event.target.value
                    })
                  }
                  placeholder="Example: Redesign login page with smooth language switch animation, but do not change backend auth API."
                  className="min-h-36 w-full resize-none rounded-xl border border-neutral-800 bg-neutral-950 p-4 text-sm leading-6 text-white outline-none transition placeholder:text-neutral-700 focus:border-neutral-600"
                />
              </div>

              <div className="grid grid-cols-2 gap-4">
                <div>
                  <label className="mb-2 block text-sm text-neutral-400">
                    Task type
                  </label>
                  <select
                    value={taskPackDraft.taskType}
                    onChange={(event) =>
                      setTaskPackDraft({
                        ...taskPackDraft,
                        taskType: event.target.value
                      })
                    }
                    className="h-11 w-full rounded-xl border border-neutral-800 bg-neutral-950 px-3 text-sm text-white outline-none focus:border-neutral-600"
                  >
                    <option value="general">General</option>
                    <option value="ui">UI / UX</option>
                    <option value="backend">Backend</option>
                    <option value="bugfix">Bugfix</option>
                    <option value="refactor">Refactor</option>
                    <option value="docs">Docs</option>
                    <option value="tests">Tests</option>
                  </select>
                </div>

                <div>
                  <label className="mb-2 block text-sm text-neutral-400">
                    Target AI tool
                  </label>
                  <select
                    value={taskPackDraft.targetTool}
                    onChange={(event) =>
                      setTaskPackDraft({
                        ...taskPackDraft,
                        targetTool: event.target.value
                      })
                    }
                    className="h-11 w-full rounded-xl border border-neutral-800 bg-neutral-950 px-3 text-sm text-white outline-none focus:border-neutral-600"
                  >
                    <option value="codex">Codex</option>
                    <option value="cursor">Cursor</option>
                    <option value="claude">Claude Code</option>
                    <option value="generic">Generic AI Agent</option>
                  </select>
                </div>
              </div>
            </div>

            <div className="flex items-center justify-end gap-3 border-t border-neutral-900 px-5 py-4">
              <button
                onClick={() => setTaskPackDraft(null)}
                className="cf-button-secondary"
              >
                Cancel
              </button>

              <button
                onClick={handleCreateTaskPack}
                disabled={isLoading || taskPackDraft.rawTask.trim().length < 3}
                className="cf-button-primary"
              >
                Generate task pack
              </button>
            </div>
          </div>
        </div>
      )}

      {generatedTaskPack && (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/70 p-8 backdrop-blur-sm">
          <div className="flex max-h-[82vh] w-full max-w-5xl flex-col overflow-hidden rounded-2xl border border-neutral-800 bg-black shadow-2xl">
            <div className="flex items-center justify-between border-b border-neutral-900 px-5 py-4">
              <div>
                <p className="text-xs uppercase tracking-[0.18em] text-neutral-600">
                  Generated Task Pack
                </p>
                <h3 className="mt-1 text-base font-medium text-white">
                  {generatedTaskPack.title}
                </h3>
              </div>

              <button
                onClick={() => setGeneratedTaskPack(null)}
                className="rounded-lg border border-neutral-800 bg-neutral-950 px-3 py-2 text-sm text-neutral-400 transition hover:text-white"
              >
                Close
              </button>
            </div>

            <pre className="flex-1 overflow-auto p-5 text-sm leading-6 text-neutral-300">
              {generatedTaskPack.generatedPrompt}
            </pre>

            <div className="flex items-center justify-end gap-3 border-t border-neutral-900 px-5 py-4">
              <button
                onClick={() => navigator.clipboard.writeText(generatedTaskPack.generatedPrompt)}
                className="cf-button-primary"
              >
                Copy prompt
              </button>
            </div>
          </div>
        </div>
      )}
    </main>
  );
}

export default App;
